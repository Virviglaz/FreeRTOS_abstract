649 template <typename... Args>
650 bool TryEmplaceBack(Args &&...args) noexcept (
651 std::is_nothrow_constructible<T, Args &&...>::value) {
652 static_assert(
653 std::is_constructible<T, Args &&...>::value,
654 "T must be constructible with Args&&...");
655
656 size_t next_wr_idx = wr_idx + 1;
657 if (next_wr_idx == size)
658 next_wr_idx = 0;
659
660 if (next_wr_idx == rd_idx)
661 return false;
662
663 new (&buffer[wr_idx]) T(std::forward<Args>(args)...);
664 semaphore.Give();
665 wr_idx = next_wr_idx;
666 return true;
667 }
668
680 bool TryPushBack(const T &item) noexcept {
681 return TryEmplaceBack(item);
682 }
683
694 T* Front(size_t wait_ms = 0) noexcept {
695 if (wr_idx == rd_idx) {
696 if (wait_ms) {
697 if (!semaphore.Take(wait_ms))
698 return nullptr;
699 } else
700 return nullptr;
701 }
702
703 return &buffer[rd_idx];
704 }
705
706 void Pop() noexcept {
707 static_assert(std::is_nothrow_destructible<T>::value,
708 "T must be nothrow destructible");
709 buffer[rd_idx].~T();
710 rd_idx++;
711 if (rd_idx == size)
712 rd_idx = 0;
713 }
