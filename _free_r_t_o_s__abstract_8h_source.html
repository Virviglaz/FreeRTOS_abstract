614 template <class T, size_t size = 2>
615 class Queue
616 {
617 protected:
618#if (configSUPPORT_STATIC_ALLOCATION == 1)
619 StaticQueue_t xStaticQueue;
620#endif /* STATIC_ALLOCATION */
621 CountingSemaphore semaphore {0, size};
622 T buffer[size];
623 size_t rd_idx = 0;
624 size_t wr_idx = 0;
625 public:
629 Queue() {}
630
634 Queue(const Queue&) = delete;
635 Queue(Queue&&) = delete;
636
647 template <typename... Args>
648 bool TryEmplaceBack(Args &&...args) noexcept (
649 std::is_nothrow_constructible<T, Args &&...>::value) {
650 static_assert(
651 std::is_constructible<T, Args &&...>::value,
652 "T must be constructible with Args&&...");
653
654 size_t next_wr_idx = wr_idx + 1;
655 if (next_wr_idx == size)
656 next_wr_idx = 0;
657
658 if (next_wr_idx == rd_idx)
659 return false;
660
661 new (&buffer[wr_idx]) T(std::forward<Args>(args)...);
662 semaphore.Give();
663 wr_idx = next_wr_idx;
664 return true;
665 }
666
678 bool TryPushBack(const T &item) noexcept {
679 return TryEmplaceBack(item);
680 }
681
692 T* Front(size_t wait_ms = 0) noexcept {
693 if (wr_idx == rd_idx) {
694 if (!semaphore.Take(wait_ms))
695 return nullptr;
696 }
697
698 return &buffer[rd_idx];
699 }
700
701 void Pop() noexcept {
702 static_assert(std::is_nothrow_destructible<T>::value,
703 "T must be nothrow destructible");
704 buffer[rd_idx].~T();
705 taskENTER_CRITICAL();
706 rd_idx++;
707 if (rd_idx == size)
708 rd_idx = 0;
709 taskEXIT_CRITICAL();
710 }
711 };
